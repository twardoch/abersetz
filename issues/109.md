Here is a smart, portable Python CLI tool built with `fire` that scans for LLM models across macOS and Windows.

This script checks the standard paths for **Hugging Face**, **Ollama**, **Pinokio**, and **LM Studio** (incorporating the `.lmstudio-home-pointer` logic). It intelligently searches for `.gguf`, `.safetensors`, `.bin`, `.pth`, `.onnx`, and `.mlpackage` formats.

### `llmfind.py`

Save the following code as `llmfind.py`.

```python
#!/usr/bin/env python3
import os
import json
from pathlib import Path
import fire

def format_size(size_bytes):
    """Convert bytes to a human-readable format."""
    if size_bytes == 0:
        return "0B"
    size_name = ("B", "KB", "MB", "GB", "TB")
    i = 0
    p = pow(1024, i)
    s = round(size_bytes / p, 2)
    while s > 1024 and i < len(size_name) - 1:
        i += 1
        p = pow(1024, i)
        s = round(size_bytes / p, 2)
    return f"{s} {size_name[i]}"

class LLMFinder:
    """
    A CLI tool to discover downloaded LLMs across various AI applications.
    """

    def __init__(self):
        self.home = Path.home()
        
        # Extensions to look for
        self.target_extensions = {
            'GGUF': ['.gguf'],
            'Safetensors': ['.safetensors'],
            'PyTorch': ['.bin', '.pt', '.pth'],
            'ONNX': ['.onnx']
        }
        
        # Bundle formats (directories)
        self.target_bundles = {
            'CoreML': ['.mlpackage']
        }

    def _get_lmstudio_path(self):
        """Determine LM Studio models path gracefully."""
        pointer_path = self.home / ".lmstudio-home-pointer"
        if pointer_path.exists():
            try:
                target_dir = pointer_path.read_text(encoding='utf-8').strip()
                return Path(target_dir) / "models"
            except Exception as e:
                print(f"[!] Error reading LMStudio pointer: {e}")
        
        # Fallbacks for Windows and Mac
        return self.home / ".cache" / "lm-studio" / "models"

    def _get_search_paths(self):
        """Return a dictionary of apps and their expected model directories."""
        paths = {
            "HuggingFace": self.home / ".cache" / "huggingface" / "hub",
            "Ollama": self.home / ".ollama" / "models",
            "LMStudio": self._get_lmstudio_path(),
            "Pinokio": self.home / "pinokio",
            "GPT4All": self.home / "AppData" / "Local" / "nomic.ai" / "GPT4All" if os.name == 'nt' else self.home / "Library" / "Application Support" / "nomic.ai" / "GPT4All"
        }
        # Only return paths that actually exist on this system
        return {app: path for app, path in paths.items() if path.exists()}

    def _get_dir_size(self, path: Path):
        """Calculate total size of a directory (used for CoreML packages)."""
        return sum(f.stat().st_size for f in path.rglob('*') if f.is_file())

    def scan(self, format=None, min_size_mb=100):
        """
        Scan for AI models.
        
        Args:
            format: (Optional) Filter by a specific extension, e.g., 'gguf' or 'safetensors'.
            min_size_mb: Minimum file size in MB to filter out small config files (default: 100).
        """
        search_paths = self._get_search_paths()
        min_size_bytes = min_size_mb * 1024 * 1024
        
        if not search_paths:
            print("No standard AI application folders found on this system.")
            return

        print(f"[*] Scanning {len(search_paths)} AI application folders...\n")

        total_models = 0
        total_size_bytes = 0

        for app_name, app_path in search_paths.items():
            print(f"--- {app_name} ({app_path}) ---")
            found_in_app = False

            for root, dirs, files in os.walk(app_path):
                root_path = Path(root)

                # 1. Check for Directory Bundles (like CoreML .mlpackage)
                # We iterate backwards to safely remove items from the list during iteration
                for i in range(len(dirs) - 1, -1, -1):
                    d = dirs[i]
                    if any(d.endswith(ext) for ext in self.target_bundles['CoreML']):
                        bundle_path = root_path / d
                        size = self._get_dir_size(bundle_path)
                        if size >= min_size_bytes:
                            print(f"  [CoreML] {d} - {format_size(size)}")
                            total_models += 1
                            total_size_bytes += size
                            found_in_app = True
                        # Don't recurse into the mlpackage bundle
                        del dirs[i]

                # 2. Check for Standard Model Files
                for f in files:
                    file_path = root_path / f
                    
                    # Special handling for Ollama (stored as extensionless sha256 blobs)
                    if app_name == "Ollama" and root_path.name == "blobs" and f.startswith("sha256-"):
                        try:
                            size = file_path.stat().st_size
                            if size >= min_size_bytes:
                                print(f"  [Ollama Blob] {f[:15]}... - {format_size(size)}")
                                total_models += 1
                                total_size_bytes += size
                                found_in_app = True
                        except OSError:
                            pass
                        continue

                    # General extension matching
                    ext = file_path.suffix.lower()
                    if format and f".{format.lower().replace('.', '')}" != ext:
                        continue # Skip if user filtered by a format and it doesn't match

                    is_model = any(ext in exts for exts in self.target_extensions.values())
                    
                    if is_model:
                        try:
                            size = file_path.stat().st_size
                            if size >= min_size_bytes:
                                print(f"  [{ext.strip('.')}] {f} - {format_size(size)}")
                                total_models += 1
                                total_size_bytes += size
                                found_in_app = True
                        except OSError:
                            pass

            if not found_in_app:
                print("  No matching models found.")
            print() # Blank line for spacing

        print(f"[*] Summary: Found {total_models} models totaling {format_size(total_size_bytes)}.")

if __name__ == '__main__':
    fire.Fire(LLMFinder)

```

### Setup & Usage

**1. Install Dependencies**
You will need `fire` installed in your Python environment:

```bash
pip install fire

```

**2. Basic Scan**
To find all models larger than 100MB across all apps (this filters out `.bin` files that are just small config payloads, focusing entirely on actual LLM weights):

```bash
python llmfind.py scan

```

**3. Filter by Format**
If you want to only find `gguf` files:

```bash
python llmfind.py scan --format=gguf

```

**4. Adjust Minimum File Size**
If you want to view smaller files or restrict it to huge models only (e.g., larger than 2GB):

```bash
python llmfind.py scan --min_size_mb=2048

```

### Why this works well:

* **Ollama Intelligence:** Ollama drops file extensions entirely and stores GGUF files in a `blobs` directory as SHA-256 hashes. The script recognizes this path hierarchy and includes large blobs as models instead of skipping them.
* **CoreML Handling:** CoreML files (`.mlpackage`) are actually macOS directories, not standard files. The script explicitly traps them during the `os.walk` directory loop, calculates the total payload size inside, and prevents unnecessary deep-scanning inside the bundle.
* **Graceful LM Studio:** It strictly follows your requirement, reading the `$HOME/.lmstudio-home-pointer` file if it exists, but will safely fallback to the standard Windows or Mac app data paths if the pointer file was deleted or cannot be read.
